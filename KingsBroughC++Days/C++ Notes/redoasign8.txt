double findlow(double a, double b,double c, double d){

double droplow=a;
   if(b<droplow)
droplow=b;
   if(c<droplow)
droplow=c;
   if (d<droplow)
droplow=d;

return droplow;

}

double findaverage(double a, double b, double c, double d){

double zlow=findlow(a,b,c,d);
average=((a+b+c+d)-zlow)/3;

}
